Table company {
  id INTEGER PK
  name TEXT [not null]
  notes TEXT
  website TEXT
}

Table role {
  id INTEGER PK
  company_id INTEGER [ref: > company.id, not null]
  title TEXT [not null]
  notes TEXT
  ad_url TEXT
  date_added DATE
}

Table contract {
  id INTEGER PK
  role_id INTEGER [ref: > role.id, not null]
  type contract_type [not null]
  salary_range NUMRANGE 
  salary_includes_super boolean
  salary_period salary_period
  salary_currency text
  term interval
}

Table role_location {
  id INTEGER PK
  role_id INTEGER [ref: > role.id, not null]
  location TEXT [not null]
  on_site boolean [not null]
  hybrid boolean [not null]
  remote boolean [not null]
  office_days NUMRANGE
}

Table requirement {
  id INTEGER PK
  role_id INTEGER [ref: > role.id, not null]
  description TEXT [not null]
  bonus boolean [not null]
}

Table application {
  id INTEGER PK
  role_id INTEGER [ref: > role.id, not null]
  cover_letter TEXT [not null]
  submitted boolean [not null]
}

Table competency {
  id INTEGER PK
  application_id INTEGER [ref: > application.id, not null]
  requirement_id INTEGER [ref: > requirement.id, not null]
  match_level requirement_match_level [not null]
  match_justification TEXT [not null]
}

Enum requirement_match_level {
  exceeded
  met
  room_for_growth
}

Enum contract_type {
  "permanent"
  "fixed_term"
}

Enum salary_period {
  year
  month
  week
  day
}

Table admin {
  id INTEGER PK
  email TEXT
  password_hash text
}

Table session {
  id TEXT PK
  expiry DATE
  admin_id INTEGER [ref: > admin.id, not null]
}
